37 /// VectorIR is a facade for the CDAG to compute the cost of the vectorization,
38 /// generate intrinsics and perform the optimizations.
39 class VectorIR {
40 public:
41  /// Types of vector operations we distinguish according their scheduling
42  enum VType {
43  /// No dependencies between operations, so it can be used a unique vector
44  /// operation
45  MAP,
46  /// Reductions imply that the operations are sequential. Nonetheless we
47  /// use this type in order to detect these special cases which can be
48  /// optimized
49  REDUCE,
50  /// Inductions, e.g. A[i] = x
51  INDUCTION,
52  /// In any other case, we just say that the vector operation must be
53  /// sequential
54  SEQ
55  };

57  /// Compute the vector width needed from the number of operands and the type
58  /// them
59  static MVDataType::VWidth getWidthFromVDataType(int NOps,
60  MVDataType::VDataType VData) {
61  int Bits = MVDataType::VDataTypeWidthBits[VData] * NOps;
62  if (Bits > 256) {
63  return MVDataType::VWidth::W512;
64  } else if (Bits > 128) {
65  return MVDataType::VWidth::W256;
66  } else if (Bits > 64) {
67  return MVDataType::VWidth::W128;
68  } else if (Bits > 32) {
69  return MVDataType::VWidth::W64;
70  } else if (Bits > 16) {
71  return MVDataType::VWidth::W32;
72  } else if (Bits > 8) {
73  return MVDataType::VWidth::W16;
74  }
75  return MVDataType::VWidth::W8;
76  }

101  /// Vector operand basically is a wrap of VL (vector length) operands in the
102  /// original Node
103  struct VOperand {
104  /// Name identifying the vector operand
105  std::string Name;
106  /// Total size of vector
107  unsigned int VSize;
108  /// Number non null Nodes
109  unsigned int Size = 4;
110  /// Array of variable size (Size elements actually) initialized when
111  /// creating the object
112  // Node **UOP = nullptr;
113  std::vector<Node *> UOP;
114  /// Data type
115  MVDataType::VDataType DType = MVDataType::VDataType::DOUBLE;
116  /// Width of this operand
117  MVDataType::VWidth Width = MVDataType::VWidth::W256;
118  /// Base array
119  std::string BaseArray = "";
120  /// All store values
121  std::vector<std::string> StoreValues;
122  /// Mask for shuffling if necessary
123  std::vector<long> Idx;
124  /// Mask to avoid elements if necessary
125  unsigned int Mask = 0x0;
126  /// High bits are loaded
127  bool HighBits = false;
128  /// Low bits are loaded
129  bool LowBits = false;
130  /// UOPS are all the same
131  bool EqualVal = true;
132  /// Memory addres is unaligned
133  bool Unaligned = false;
134  /// Memory is contiguous
135  bool Contiguous = true;
136  /// Is partial (mask is not all 1)
137  bool IsPartial = false;
138  /// Values are in the same vector
139  bool SameVector = true;
140  /// If the memory addresses cross cache line sizes
141  bool OutOfCacheLine = false;
142  /// The vector operand is a temporal result if it has already been assigned
143  /// before
144  bool IsTmpResult = false;
145  /// Vector operand could be a load if it is not a temporal result
146  bool IsLoad = false;
147  /// Vector UOPs are memory addresses
148  bool MemOp = false;
149  /// Vector operand may be a store in memory operation
150  bool IsStore = false;
151  /// Order of the vector operation
152  int Order = -1;
153  /// Offset of the vector operation
154  int Offset = -1;
155  /// Check if there is a vector already assigned wraping the same values
156  bool checkIfVectorAssigned(int VL, Node::NodeListType &V,
157  MVDataType::VWidth W);
158  /// Get width
159  MVDataType::VWidth getWidth() { return Width; }
160  /// Get data type of the operand: assumption that all elements are the same
161  /// type
162  MVDataType::VDataType getDataType() {
163  // Be careful with this
164  return DType;
165  }
166  /// Return name of VOperand
167  std::string getName() { return this->Name; }
168 
169  /// Returns if operands have been already loaded from memory
170  bool checkIfAlreadyLoaded(Node *PrimaryNode);
171 
172  /// Return if operands already stored in memory
173  bool checkIfAlreadyStored(Node::NodeListType &V);
174 
175  /// Generate a name according to the VID
176  std::string genNewVOpName() { return VOP_PREFIX + std::to_string(VID++); }
177 
178  /// Return register name
179  std::string getRegName(int Position, int Offset) {
180  if (Offset != 0) {
181  auto Operand = this->UOP[Position]->getMVExpr();
182  if (Operand != nullptr) {
183  auto MVE = dyn_cast<MVExprArray>(Operand);
184  return MVE->toStringWithOffset(Offset);
185  } else {
186  MVErr("Something went wrong...");
187  }
188  }
189  return this->UOP[Position]->getRegisterValue();
190  }
191 
192  /// Get loop where the result is computed
193  std::string getOperandLoop() {
194  return (UOP[0] != nullptr) ? UOP[0]->getLoopName() : "";
195  }
196 
197  /// Copy constructor
198  VOperand(const VOperand &V) {
199  this->Name = V.Name;
200  this->BaseArray = V.BaseArray;
201  this->VSize = V.VSize;
202  this->Size = V.Size;
203  this->UOP = V.UOP;
204  this->StoreValues = V.StoreValues;
205  this->DType = V.DType;
206  this->Width = V.Width;
207  this->Idx = V.Idx;
208  this->Mask = V.Mask;
209  this->EqualVal = V.EqualVal;
210  this->Unaligned = V.Unaligned;
211  this->Contiguous = V.Contiguous;
212  this->IsPartial = V.IsPartial;
213  this->HighBits = V.HighBits;
214  this->LowBits = V.LowBits;
215  this->SameVector = V.SameVector;
216  this->OutOfCacheLine = V.OutOfCacheLine;
217  this->IsTmpResult = V.IsTmpResult;
218  this->IsLoad = V.IsLoad;
219  this->MemOp = V.MemOp;
220  this->IsStore = V.IsStore;
221  this->Order = V.Order;
222  this->Offset = V.Offset;
223  }
224 
225  /// Printing the vector operand
226  std::string toString();
227 
228  /// Basic constructor
229  VOperand(int VL, Node::NodeListType &V, bool Res);
230 
231  /// Empty constructor
232  VOperand();
233  };

235  /// Main component of the VectorIR which wraps the selected DAGs based on
236  /// the placement and/or free scheduling. Basically represents the
237  /// following: VectorOP -> VOperand = VectorOP[VOperand,
238  /// VOperand]; Where each OPS is a VOperand, which basically wraps
239  /// a list of UnitOperands or Nodes
240  struct VectorOP {
241  /// Type of operation
242  VType VT = VType::MAP;
243  /// Name of the vector operation
244  std::string VN = "";
245  /// Vector Width of data used in this operation
246  MVDataType::VWidth VW = MVDataType::VWidth::W256;
247  /// Vector data type used
248  MVDataType::VDataType DT = MVDataType::VDataType::DOUBLE;
249  /// Vector first operand
250  VOperand OpA;
251  /// Vector second operand
252  VOperand OpB;
253  /// If there is only one operand, then it is unary
254  bool IsUnary = false;
255  /// Vector result
256  VOperand R;
257  /// Order of the vector operation
258  int Order = -1;
259  /// Offset regarding the original order
260  int Offset = -1;
261  /// Cost associated to this concrete VectorOP
262  unsigned int Cost = 0;
263  /// Check if it is a BinaryOperation
264  bool isBinOp();
265  /// Get the BinaryOperator::Opcode
266  BinaryOperator::Opcode getBinOp();
267  /// Get the MVOp
268  MVOp getMVOp();
269  /// Constructor from the CDAG
270  VectorOP(int VL, Node::NodeListType &VOps, Node::NodeListType &VLoadA,
271  Node::NodeListType &VLoadB);
272  /// Check if operation is sequential or not
273  bool isSequential() { return this->VT == VType::SEQ; }
274  /// Render vector operation as string
275  std::string toString();
276  };
277 };
