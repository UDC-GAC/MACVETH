43 /// The Location of the Scop, as delimited by macveth and endmacveth
44 /// pragmas by the user.
45 /// "macveth" and "endmacveth" are the source locations of the macveth and
46 /// endmacveth pragmas.
47 /// "StartLine" is the Line number of the Start position.
48 struct ScopLoc {
49  ScopLoc() : End(0) {}
50 
51  /// File identifier of the scop
52  FileID FID;
53  /// Start location of the pragma
54  clang::SourceLocation Scop;
55  /// End location of the pragma
56  clang::SourceLocation EndScop;
57  unsigned StartLine = 0;
58  unsigned EndLine = 0;
59  unsigned Start = 0;
60  unsigned End = 0;
61  std::list<std::string> DimVisited = {};
62  bool ScopHasBeenScanned = false;

64  struct PragmaArgs {
65  /// Generate SIMD code in the region: true by default
66  bool SIMDCode = true;
67  /// General option for unrolling code
68  bool Unroll = true;
69  /// Unroll factor (general)
70  int UnrollFactor = 1;
71  /// Perform unroll and jam
72  bool UnrollAndJam = true;
73  /// Full unrolling
74  std::map<std::string, bool> FullUnroll;
75  /// Custom option for each loop within the scope
76  PragmaTupleDim UnrollDim;
77  };

79  /// Each scop has its own PragmaArgs, which are parsed at pre-processing time
80  PragmaArgs PA;

83 /// List of pairs of #pragma macveth and #pragma endmacveth source locations
84 struct ScopHandler {
85  static inline std::vector<ScopLoc *> List;

90  /// Return a SourceLocation for Line "Line", column "col" of file "FID".
91  static SourceLocation translateLineCol(SourceManager &SM, FileID FID,
92  unsigned Line, unsigned col) {
93  return SM.translateLineCol(FID, Line, col);

96  /// Update scop to visited
97  static void visitScop(ScopLoc S) {
98  for (auto SV : List) {
99  if (SV->StartLine == S.StartLine) {
100  SV->ScopHasBeenScanned = true;
101  return;
102  }
103  }
104  }

106  /// Get ROI of a concrete function
107  static std::vector<ScopLoc *> funcGetScops(FunctionDecl *fd) {
108  SourceManager &SM = fd->getParentASTContext().getSourceManager();
109  unsigned int StartFD = SM.getExpansionLineNumber(fd->getBeginLoc());
110  unsigned int EndFD = SM.getExpansionLineNumber(fd->getEndLoc());
111  std::vector<ScopLoc *> SList;
112  for (auto SL : List) {
113  auto FunctionFileID = SM.getFileID(fd->getLocation());
114  if (SL->FID != FunctionFileID) {
115  continue;
116  }
117  unsigned int StartL = SM.getFileOffset(
118  translateLineCol(SM, SM.getFileID(SL->Scop), StartFD, 1));
119  unsigned int EndL = SM.getFileOffset(
120  translateLineCol(SM, SM.getFileID(SL->EndScop), EndFD + 1, 1));
121 
122  if (((StartFD <= SL->StartLine) && (EndFD >= SL->EndLine)) &&
123  ((StartL <= SL->Start) && (EndL >= SL->End)))
124  SList.push_back(SL);
125  }
126  return SList;
127  }
128 
129  /// Determine whether a function has or not ROI within
130  static bool funcHasROI(FunctionDecl *fd) {
131  return (funcGetScops(fd).size() > 0);
132  }
133 
134  /// Add a new Start (#pragma macveth) Location to the list.
135  /// If the last #pragma macveth did not have a matching
136  /// #pragma endmacveth then overwrite it.
137  /// "Start" points to the location of the macveth pragma.
138  void addStart(SourceManager &SM, SourceLocation Start,
139  ScopLoc::PragmaArgs PA) {
140  ScopLoc *Loc = new ScopLoc();
141  MACVETH_DEBUG("MVPragmaHandler", "addStart");
142  Loc->FID = SM.getFileID(Start);
143  Loc->PA = PA;
144  Loc->Scop = Start;
145  int Line = SM.getExpansionLineNumber(Start);
146  Loc->StartLine = Line;
147  Loc->Start = SM.getFileOffset(translateLineCol(SM, Loc->FID, Line, 1));
148  if (List.size() == 0 || List[List.size() - 1]->End != 0) {
149  List.push_back(Loc);
150  } else {
151  List[List.size() - 1] = Loc;
152  }
153  }
154 
155  /// Set the end location (#pragma endmacveth) of the last pair
156  /// in the list.
157  /// If there is no such pair of if the end of that pair
158  /// is already set, then ignore the spurious #pragma endmacveth.
159  /// "end" points to the location of the endmacveth pragma.
160  void addEnd(SourceManager &SM, SourceLocation End) {
161  if (List.size() == 0 || List[List.size() - 1]->End != 0)
162  return;
163  List[List.size() - 1]->EndScop = End;
164  int Line = SM.getExpansionLineNumber(End);
165  End = translateLineCol(SM, SM.getFileID(End), Line + 1, 1);
166  List[List.size() - 1]->End = SM.getFileOffset(End);
167  List[List.size() - 1]->EndLine = Line;
168  }
169 };

171 /// Get IdentifierInfo from Token
172 static IdentifierInfo *getValue(Token &token) {
173  if (token.isNot(tok::identifier)) {
174  return nullptr;
175  }
176  if (token.getKind() == tok::eod) {
177  return nullptr;
178  }
179  return token.getIdentifierInfo();
180 }

182 /// Parse arguments using the preprocessor
183 static ScopLoc::PragmaArgs parseArguments(Preprocessor &PP) {
184  ScopLoc::PragmaArgs PA;
185  IdentifierInfo *II;
186  Token Tok;
187  PP.Lex(Tok);
188  bool UnrollOptParsed = false;
189  while ((II = getValue(Tok)) != nullptr) {
190  PP.Lex(Tok);
191  if (((II->isStr("nounroll")) || (II->isStr("unroll")) ||
192  (II->isStr("unrollandjam"))) &&
193  (UnrollOptParsed)) {
194  assert(false &&
195  "You can not specify twice unrolling options in the same scop!");
196  }
197 
198  // Check unroll options
199  if (II->isStr("nounroll") || II->isStr("unrollandjam") ||
200  II->isStr("unroll")) {
201  PA.Unroll = (II->isStr("unrollandjam") || II->isStr("unroll"));
202  PA.UnrollAndJam = (II->isStr("unrollandjam"));
203  UnrollOptParsed = true;
204  continue;
205  }
206 
207  // Check whether MACVETH should emit SIMD code or just scalar code
208  if (II->isStr("scalar") || II->isStr("nosimd")) {
209  PA.SIMDCode = false;
210  continue;
211  }
212 
213  // Otherwise it will be a unrolling dimension
214  if ((!Tok.isLiteral())) {
215  auto IINext = getValue(Tok);
216  if (IINext == nullptr) {
217  MVErr("Pragma bad formatting");
218  }
219  if (IINext->isStr("full")) {
220  PA.FullUnroll[II->getName().str()] = true;
221  PA.UnrollDim.push_back(
222  std::tuple<std::string, int>(II->getName().str(), 1));
223  PP.Lex(Tok);
224  continue;
225  } else {
226  MVErr("Pragmas bad syntax...");
227  }
228  }
229 
230  auto IINext = Tok.getLiteralData();
231  if (IINext == nullptr) {
232  assert(false &&
233  "Bad format: needed a unrolling factor for the dimension");
234  }
235 
236  int UnrollFactor = atoi(IINext);
237  if (UnrollFactor == 0) {
238  assert(false && "Bad value for unrolling factor");
239  }
240  PA.UnrollDim.push_back(
241  std::tuple<std::string, int>(II->getName().str(), UnrollFactor));
242  PP.Lex(Tok);
243  }
244  return PA;
245 }

247 /// Handle pragmas of the form
248 ///
249 /// #pragma macveth [unroll|nounroll|unrollandjam] [loopIdentifier
250 /// unrollFactor] [nosimd]
251 ///
252 /// In particular, store the Location of the line containing
253 /// the pragma in the list "Scops". The intention of this pragma is to tell the
254 /// compiler where the ROI starts.
255 struct PragmaScopHandler : public PragmaHandler {
256  ScopHandler *Scops;
257 
258  PragmaScopHandler(ScopHandler *Scops)
259  : PragmaHandler("macveth"), Scops(Scops) {}
260 
261  virtual void HandlePragma(Preprocessor &PP, PragmaIntroducer Introducer,
262  Token &ScopTok) {
263  SourceManager &SM = PP.getSourceManager();
264  SourceLocation Sloc = ScopTok.getLocation();
265  Scops->addStart(SM, Sloc, parseArguments(PP));
266  }
267 };
268 
269 /// Handle pragmas of the form
270 ///
271 /// #pragma endmacveth
272 ///
273 /// In particular, store the Location of the line following the one containing
274 /// the pragma in the list "Scops". The intention of this pragma is to tell the
275 /// compiler where the ROI ends.
276 struct PragmaEndScopHandler : public PragmaHandler {
277  ScopHandler *Scops;
278 
279  PragmaEndScopHandler(ScopHandler *Scops)
280  : PragmaHandler("endmacveth"), Scops(Scops) {}
281 
282  virtual void HandlePragma(Preprocessor &PP, PragmaIntroducer Introducer,
283  Token &EndScopTok) {
284  SourceManager &SM = PP.getSourceManager();
285  SourceLocation Sloc = EndScopTok.getLocation();
286  Scops->addEnd(SM, Sloc);
287  }
288 };
