55 /// Wrapper for calculating the cost of instructions
56 struct InstCostInfo {
57  /// Default empty constructor
58  InstCostInfo() {}
59 
60  /// Constructor with all parameters
61  InstCostInfo(int L, float T, int U) : Latency(L), Throughput(T), UOps(U) {}
62 
63  /// Default constructor given a row from the CostTable and the type of
64  /// instruction
65  InstCostInfo(CostTable::Row R, InstType I) {
66  UOps = R.NUops;
67  if (I == InstType::SIMDOp) {
68  Latency = CostTable::getLatencyIntrin(R.Intrinsics);
69  } else if (I == InstType::MVOP) {
70  Latency = CostTable::getLatencyMVOP(R.MVOP);
71  }
72  Throughput = R.Throughput;
73  }
74 
75  /// Number of cycles wasted by the instruction
76  int Latency = 0;
77  /// TODO: Issue latency: the lower the better. On Intel architectures it
78  /// represents the velocity of issuing another uop
79  double Throughput = 1.0;
80  /// Number of micro-instructions issued by the SIMD instruction
81  int UOps = 1;
82  /// TODO: Number of ports used
83  std::map<int, int> NoPorts;
84 
85  std::string toString() { return std::to_string(Latency); }
86 
87  long operator-(const InstCostInfo &R) { return this->Latency - R.Latency; }
88 
89  InstCostInfo operator+=(const InstCostInfo &R) {
90  this->Latency += R.Latency;
91  return *this;
92  }
93 
94  InstCostInfo operator+(const InstCostInfo &R) {
95  this->Latency += R.Latency;
96  return *this;
97  }
98 
99  bool operator<(const InstCostInfo &R) { return this->Latency < R.Latency; }
100 
101  bool operator<=(const InstCostInfo &R) { return this->Latency <= R.Latency; }
102 };

131  /// Generate cost report
132  void generateCostReport() {
133  if (MVOptions::SIMDReportFile == "") {
134  return;
135  }
136  std::ofstream F;
137  F.open(MVOptions::SIMDReportFile);
138  if (!F.is_open()) {
139  MVWarn("Cost report file could not be opened!");
140  return;
141  }
142  F << " === MACVETH: SIMD report" << std::endl;
143  for (auto It = CostOp.begin(); It != CostOp.end(); ++It) {
144  F << It->first + " \t = \t" +
145  std::to_string(NumOp[It->first] * It->second.Latency) + "\t(" +
146  std::to_string(NumOp[It->first]) + ")"
147  << std::endl;
148  }
149  F << " TOTAL COST = " + std::to_string(TotCost) << std::endl;
150  F << " === ENDMACVETH" << std::endl;
151  F.close();
152  }
153 };
