76 /// MVOp is meant to identify those operations that Clang lacks. This way
77 /// parsing is easier for backends and more extensible. It is implemented using
78 /// a 'union' over 'std::variant' (C++14) since both possible types are enums
79 /// and may have the same values; this way we can explicitly handle each case
80 struct MVOp {
81  /// Prefix
82  std::string Prefix = "";
83  /// Type of the MVOp
84  MVOpType T;
85  /// We can handle Clang BinOps or MVOps
86  union {
87  /// When CLANG_BINOP
88  clang::BinaryOperator::Opcode ClangOP;
89  /// When MVFUNC
90  MVOpCode MVOpC;
91  };
92 
93  /// Given a string it returns the type of function based on the MVOpCode it
94  /// is defined for this compiler
95  static MVOpCode getTypeFromStr(std::string S) {
96  if ((S == "std::max") || (S == "max") || (S == "MAX"))
97  return MVOpCode::MAX;
98  if ((S == "std::min") || (S == "min") || (S == "MIN"))
99  return MVOpCode::MIN;
100  if ((S == "avg") || (S == "AVG"))
101  return MVOpCode::AVG;
102  if ((S == "round") || (S == "ROUND"))
103  return MVOpCode::ROUND;
104 
105  MVAssertSkip(MVOptions::IntrinsicsSVML,
106  "No SVML, so function " + S + " can not be vectorized",
107  GotoStartScop);
108 
109  if ((S == "ceil") || (S == "CEIL"))
110  return MVOpCode::CEIL;
111  if ((S == "floor") || (S == "FLOOR"))
112  return MVOpCode::FLOOR;
113  if ((S == "exp") || (S == "EXP"))
114  return MVOpCode::EXP;
115  if ((S == "pow") || (S == "powf") || (S == "powl"))
116  return MVOpCode::POW;
117  if ((S == "sin") || (S == "SIN"))
118  return MVOpCode::SIN;
119  if ((S == "cos") || (S == "COS"))
120  return MVOpCode::COS;
121 
122  return MVOpCode::UNDEF;
123  }

125  /// Return the equivalent string given a MVOpCode
126  std::string getStrFromMVCode(MVOpCode C) {
127  switch (C) {
128  case MVOpCode::ADD:
129  return "add";
130  case MVOpCode::SUB:
131  return "sub";
132  case MVOpCode::DIV:
133  return "div";
134  case MVOpCode::MUL:
135  return "mul";
136  case MVOpCode::EXP:
137  return "exp";
138  case MVOpCode::MAX:
139  return "max";
140  case MVOpCode::MIN:
141  return "min";
142  case MVOpCode::AVG:
143  return "avg";
144  case MVOpCode::CEIL:
145  return "ceil";
146  case MVOpCode::FLOOR:
147  return "floor";
148  case MVOpCode::ROUND:
149  return "round";
150  case MVOpCode::SIN:
151  return "sin";
152  case MVOpCode::COS:
153  return "cos";
154  default:
155  return "undef";
156  }
157  return "undef";
158  }

160  /// Return the equivalent string given a MVOpCode
161  std::string
162  getStrFromBinaryOperatorOpcode(clang::BinaryOperator::Opcode ClangOP) {
163  switch (ClangOP) {
164  case clang::BinaryOperator::Opcode::BO_Add:
165  return "add";
166  case clang::BinaryOperator::Opcode::BO_Sub:
167  return "sub";
168  case clang::BinaryOperator::Opcode::BO_Div:
169  return "div";
170  case clang::BinaryOperator::Opcode::BO_Mul:
171  return "mul";
172  default:
173  return "undef";
174  }
175  return "undef";
176  }

178  /// Shortcut to check whether a MVOp is of type BO_Assign
179  bool isAssignment() {
180  return (getType() == MVOpType::CLANG_BINOP) &&
181  (this->ClangOP == clang::BinaryOperator::Opcode::BO_Assign);
182  }
183 
184  /// Check whether the operation is binary or not. An operation is binary if
185  /// it is placed between two operands, in other case if is a function
186  bool isBinaryOperation() { return (getType() == MVOpType::CLANG_BINOP); }
187 
188  /// Check whether the operation is an addition or a substraction. This is
189  /// useful for reductions
190  bool isAddOrSub() {
191  return (this->ClangOP == BinaryOperator::Opcode::BO_Add) ||
192  (this->ClangOP == BinaryOperator::Opcode::BO_Sub);
193  }
194 
195  /// Get equivalent MVOP string for searching in the Table Cost
196  std::string getTableMVOPstr(std::string T) {
197  std::string Suffix = "_" + Utils::toUppercase(T);
198  if (getType() == MVOpType::CLANG_BINOP) {
199  return Utils::toUppercase(getStrFromBinaryOperatorOpcode(this->ClangOP)) +
200  Suffix;
201  }
202  return Utils::toUppercase(getStrFromMVCode(this->MVOpC)) + Suffix;
203  }
204 
205  /// Get the namespace of the expression
206  std::string getPrefixFromStr(std::string S) {
207  std::string P = "";
208  std::string NameSpace = "::";
209  auto FoundNameSpace = S.find(NameSpace);
210  if (FoundNameSpace != std::string::npos) {
211  P = S.substr(0, FoundNameSpace + NameSpace.size());
212  }
213  return P;
214  }
215 
216  /// Convert MVOp to string
217  std::string toString() {
218  if (getType() == MVOpType::CLANG_BINOP) {
219  return clang::BinaryOperator::getOpcodeStr(this->ClangOP).str();
220  }
221  return getStrFromMVCode(this->MVOpC);
222  }
223 
224  /// Get the prefix
225  std::string getOpPrefix() { return this->Prefix; }
226 
227  /// Return the type of the operation
228  MVOpType getType() { return T; }
229 
230  /// Constructor given a string: will be a MVFUNC
231  MVOp(std::string S) {
232  this->T = MVOpType::MVFUNC;
233  this->MVOpC = getTypeFromStr(S);
234  this->Prefix = getPrefixFromStr(S);
235  }
236 
237  /// Given a BinaryOperator::Opcode the MVOp will be of CLANG_BINOP type
238  MVOp(clang::BinaryOperator::Opcode OP) {
239  this->T = MVOpType::CLANG_BINOP;
240  this->ClangOP = OP;
241  }
242 
243  /// Empty constructor: UNDEF MVOp
244  MVOp() {
245  this->T = MVOpType::MVFUNC;
246  this->MVOpC = MVOpCode::UNDEF;
247  }
248 };
