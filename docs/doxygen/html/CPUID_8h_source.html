45 class CPUID {
46  /// EAX, EBX, ECX, EDX
47  uint32_t Regs[4];
48 
49 public:
50  explicit CPUID(unsigned FuncId, unsigned SubFuncId) {
51  // ECX is set to zero for CPUID function 4
52 #ifdef _WIN32
53  __cpuidex((int *)Regs, (int)FuncId, (int)SubFuncId);
54 #else
55  asm volatile("cpuid"
56  : "=a"(Regs[0]), "=b"(Regs[1]), "=c"(Regs[2]), "=d"(Regs[3])
57  : "a"(FuncId), "c"(SubFuncId));
58 #endif
59  }
60 
61  /// Return EAX register
62  const uint32_t &EAX() const { return Regs[0]; }
63  /// Return EBX register
64  const uint32_t &EBX() const { return Regs[1]; }
65  /// Return ECX register
66  const uint32_t &ECX() const { return Regs[2]; }
67  /// Return EDX register
68  const uint32_t &EDX() const { return Regs[3]; }
69 };

128  MVCPUInfo() {
129  // Get vendor name EAX=0
130  CPUID CpuID0(0, 0);
131  uint32_t HFS = CpuID0.EAX();
132  VendorId += std::string((const char *)&CpuID0.EBX(), 4);
133  VendorId += std::string((const char *)&CpuID0.EDX(), 4);
134  VendorId += std::string((const char *)&CpuID0.ECX(), 4);
135 
136  // Get processor brand std::string
137  // This seems to be working for both Intel & AMD vendors
138  for (auto i = 0x80000002; i < 0x80000005; ++i) {
139  CPUID CpuID(i, 0);
140  ModelName += std::string((const char *)&CpuID.EAX(), 4);
141  ModelName += std::string((const char *)&CpuID.EBX(), 4);
142  ModelName += std::string((const char *)&CpuID.ECX(), 4);
143  ModelName += std::string((const char *)&CpuID.EDX(), 4);
144  }
145 
146  // Get SSE instructions availability
147  CPUID CpuID1(1, 0);
148  ArchName = getArchitectureFromCPUID(CpuID1);
149  IsHTT = CpuID1.EDX() & AVX_POS;
150  IsSSE = CpuID1.EDX() & SSE_POS;
151  IsSSE2 = CpuID1.EDX() & SSE2_POS;
152  IsSSE3 = CpuID1.ECX() & SSE3_POS;
153  IsSSE41 = CpuID1.ECX() & SSE41_POS;
154  IsSSE42 = CpuID1.ECX() & SSE41_POS;
155  IsAVX = CpuID1.ECX() & AVX_POS;
156  // Get AVX2 instructions availability
157  CPUID CpuID7(7, 0);
158  IsAVX2 = CpuID7.EBX() & AVX2_POS;
159  // Get AVX512 instructions availability EBX
160  IsAVX512F = CpuID7.EBX() & AVX512F_POS;
161  IsAVX512DQ = CpuID7.EBX() & AVX512DQ_POS;
162  IsAVX512IFMA = CpuID7.EBX() & AVX512IFMA_POS;
163  IsAVX512BW = CpuID7.EBX() & AVX512BW_POS;
164  IsAVX512VL = CpuID7.EBX() & AVX512VL_POS;
165  // Get AVX512 instructions availability ECX
166  IsAVX512VBMI = CpuID7.ECX() & AVX512VBMI_POS;
167  IsAVX512VBMI2 = CpuID7.ECX() & AVX512VBMI2_POS;
168  IsAVX512VNNI = CpuID7.ECX() & AVX512VNNI_POS;
169  IsAVX512BITALG = CpuID7.ECX() & AVX512BITALG_POS;
170  // Get AVX512 instructions availability EDX
171  IsAVX5124VNNIW = CpuID7.EDX() & AVX5124VNNIW_POS;
172  IsAVX5124FMAPS = CpuID7.EDX() & AVX5124VFMAPS_POS;
173  IsAVX512VP2INTERSECT = CpuID7.EDX() & IsAVX512VP2INTERSECT;
174 
175  CPUID CpuID71(7, 1);
176  IsAVX512VP2INTERSECT = CpuID71.EAX() & AVX512BF16_POS;
177 
178  std::string UpVId = VendorId;
179  for_each(UpVId.begin(), UpVId.end(), [](char &in) { in = ::toupper(in); });
180  // Get num of cores
181  if (UpVId.find("INTEL") != std::string::npos) {
182  if (HFS >= 11) {
183  for (auto lvl = 0; lvl < MAX_INTEL_TOP_LVL; ++lvl) {
184  CPUID CpuID4(0x0B, lvl);
185  uint32_t currLevel = (LVL_TYPE & CpuID4.ECX()) >> 8;
186  switch (currLevel) {
187  case 0x01:
188  NumSMT = LVL_CORES & CpuID4.EBX();
189  break;
190  case 0x02:
191  NumLogCpus = LVL_CORES & CpuID4.EBX();
192  break;
193  default:
194  break;
195  }
196  }
197  NumCores = NumLogCpus / NumSMT;
198  } else {
199  if (HFS >= 1) {
200  NumLogCpus = (CpuID1.EBX() >> 16) & 0xFF;
201  if (HFS >= 4) {
202  NumCores = 1 + ((CPUID(4, 0).EAX() >> 26) & 0x3F);
203  }
204  }
205  if (IsHTT) {
206  if (!(NumCores > 1)) {
207  NumCores = 1;
208  NumLogCpus = (NumLogCpus >= 2 ? NumLogCpus : 2);
209  }
210  } else {
211  NumCores = NumLogCpus = 1;
212  }
213  }
214  } else if (UpVId.find("AMD") != std::string::npos) {
215  if (HFS >= 1) {
216  NumLogCpus = (CpuID1.EBX() >> 16) & 0xFF;
217  if (CPUID(0x80000000, 0).EAX() >= 8) {
218  NumCores = 1 + (CPUID(0x80000008, 0).ECX() & 0xFF);
219  }
220  }
221  if (IsHTT) {
222  if (!(NumCores > 1)) {
223  NumCores = 1;
224  NumLogCpus = (NumLogCpus >= 2 ? NumLogCpus : 2);
225  }
226  } else {
227  NumCores = NumLogCpus = 1;
228  }
229  } else {
230  // FIXME: do something for ARM for instance...
231  }
232  }
