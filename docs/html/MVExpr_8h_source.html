44 /// Expressions in MACVETH inherit from MVExpr
45 ///
46 /// This class holds a double identity for expressions: a string name (for
47 /// debugging purposes and for clarity) and the actual Clang Expr, for
48 /// transformations. Besides, it also "simplifies" the typing of the expression
49 /// to some defined types (MVExprType). This will help to calculate offsets
50 /// and to simplify the Clang AST typing.
51 class MVExpr {
52 public:
53  /// Types of MVExpr
54  enum MVExprKind {
55  /// Defined in MVExprArray
56  MVK_Array,
57  /// Defined in MVExprLiteral
58  MVK_Literal,
59  /// Defined in MVExprVar
60  MVK_Var,
61  /// Other
62  MVK_Other
63  };
64 
65  /// A MVExpr can be created from a clang::Expr or from a string, it can be an
66  /// array a literal or even a variable, but when it comes to unroll, it is not
67  /// the same if it is a temporal expression created ad-hoc (e.g. tempX
68  /// register which holds a temporal value in the TAC representation). This
69  /// way, we can distinguish those cases.
70  enum MVExprInfo { EXPR_CLANG, TAC_EXPR, TMP_RES };
71 
72  /// Destructor
73  virtual ~MVExpr(){};
74 
75  /// Clone constructor
76  MVExpr(MVExpr *TE) {
77  this->setExprStr(TE->getExprStr());
78  this->setClangExpr(TE->getClangExpr());
79  this->setTypeStr(TE->getTypeStr());
80  this->setTempInfo(TE->getTempInfo());
81  this->setKind(TE->getKind());
82  }
83 
84  /// Create MVExpr from string. In this case we will assume we are treating
85  /// with a TAC_EXPR
86  MVExpr(MVExprKind K, std::string E) : MK(K), ExprStr(E) {
87  this->TempInfo = TAC_EXPR;
88  }
89 
90  /// Create MVExpr from clang::Expr, which sets its inner type/info to
91  /// EXPR_CLANG
92  MVExpr(MVExprKind K, Expr *E) : MK(K), ClangExpr(E) {
93  this->TempInfo = MVExprInfo::EXPR_CLANG;
94  this->ExprStr = Utils::getStringFromExpr(E);
95  if (isa<clang::TypedefType>(E->getType())) {
96  auto T = dyn_cast<clang::TypedefType>(E->getType());
97  if (!T->isSugared()) {
98  MACVETH_DEBUG(
99  "MVExpr",
100  "This is not 'sugared'... That is an error... Unreachable");
101  llvm::llvm_unreachable_internal();
102  }
103  auto ET = dyn_cast<ElaboratedType>(T->desugar());
104  auto TT = dyn_cast<TypedefType>(ET->desugar());
105  auto Subs = dyn_cast<SubstTemplateTypeParmType>(TT->desugar());
106  this->setTypeStr(Subs->desugar().getAsString());
107  } else {
108  this->setTypeStr(E->getType().getAsString());
109  }
110  this->MVDT = MVDataType::CTypeToVDataType[this->getTypeStr()];
111  }
112 
113  /// Set info
114  void setTempInfo(MVExprInfo TempInfo) { this->TempInfo = TempInfo; }
115 
116  /// Get MVExpr info
117  MVExprInfo getTempInfo() { return this->TempInfo; }
118 
119  /// Set clang::Expr attribute
120  void setClangExpr(clang::Expr *ClangExpr) { this->ClangExpr = ClangExpr; }
121 
122  /// Get clang::Expr attribute
123  clang::Expr *getClangExpr() const { return this->ClangExpr; }
124 
125  /// Set type as a string
126  void setTypeStr(std::string TypeStr) {
127  // This is a fucking hack: treat const as if they are not const...
128  // Thus, typing is easier. Do we really need to know if a variable is
129  // constant or not? I think we do not.
130  auto LastNotNullptrToken = TypeStr;
131  char *dup = strdup(TypeStr.c_str());
132  auto Tok = strtok(dup, " ");
133  while (Tok != nullptr) {
134  LastNotNullptrToken = Tok;
135  Tok = strtok(nullptr, " ");
136  }
137  this->TypeStr = LastNotNullptrToken;
138  // At least be clean
139  free(dup);
140  }
141 
142  /// Get type as a string
143  std::string getTypeStr() { return this->TypeStr; }
144 
145  /// Set expression as an string
146  void setExprStr(std::string ExprStr) { this->ExprStr = ExprStr; }
147 
148  /// To string function
149  virtual std::string toString() const = 0;
150 
151  /// Get expression as an string
152  std::string getExprStr() const { return this->toString(); }
153 
154  /// Get the expression as a "raw" string. Should avoid this if the object is
155  /// volatile meaning for that it _can_ change, e.g. a[i] when parsed can be
156  /// converted as a string "a[i]". However, if we unroll the value then we
157  /// will still have "a[i]" instead of "a[i+unroll]". That is why toString()
158  /// is way more powerful
159  std::string getExprStrRaw() const { return this->ExprStr; }
160 
161  /// Set kind of the expression
162  void setKind(MVExprKind Kind) { this->MK = Kind; }
163 
164  /// Get kind of the expression
165  MVExprKind getKind() const { return this->MK; }
166 
167  /// This method allows to check whether the expression belongs or not to the
168  /// original AST created by Clang
169  bool isNotClang() { return (this->getTempInfo() != EXPR_CLANG); }
170 
171  /// Given a MVExpr it will return its unrolled version
172  virtual MVExpr *unrollExpr(int UF, std::string LL,
173  bool SubstituteVal = false) = 0;
174 
175  /// Given a MVExpr and the map of unrolled,loop_level, it will return the
176  /// unrolled version
177  virtual MVExpr *unrollExpr(std::unordered_map<int, std::string> LList) {
178  return this;
179  }
180 
181  /// If this function is called, then the expression is in memory
182  void setLoadFromMem() { this->NeedsMemLoad = true; }
183 
184  /// If this function is called, then the expression is not in memory
185  void setDoNotLoadFromMem() { this->NeedsMemLoad = false; }
186 
187  /// Check if expression is in memory or not
188  bool needsToBeLoaded() { return this->NeedsMemLoad; }
189 
190  /// Two expressions are equal if and only if their name or original code
191  /// expression is equal, regardless where they appear in code.
192  bool operator==(const MVExpr &MVE) const {
193  return !getExprStr().compare(MVE.getExprStr());
194  }
195 
196  /// Two expressions are not equal if and only if their name or original code
197  /// expression is not equal.
198  bool operator!=(MVExpr &MVE) { return !operator==(MVE); }
199 
200  /// Operators
201  virtual int operator-(const MVExpr &MVE) { return -1; }
202 
203 private:
204  /// MVExpr kind
205  MVExprKind MK;
206  /// Type of MVExpr
207  MVExprInfo TempInfo = TMP_RES;
208  /// Type of data in string
209  std::string TypeStr = "";
210  /// MACVETH's datatype
211  MVDataType::VDataType MVDT = MVDataType::VDataType::UNDEF;
212  /// Expresion as Clang's
213  clang::Expr *ClangExpr = nullptr;
214  /// Expression as string
215  std::string ExprStr = "";
216  /// Need to be loaded from mem
217  bool NeedsMemLoad = true;
218  /// Dimensions of the expressions
219  std::list<std::string> Dims;
220 };
221
