40 /// All Nodes belong to a CDAG. Each node or vertex holds information regarding
41 /// the type of operation as well as its Edges (or links to another Nodes).
42 /// Nodes also hold information regarding the scheduling, i.e. the order on
43 /// which they are consumed.
44 class Node {
45 public:
46  /// Simply universal identifier for each instance node. We have to be cautious
47  /// with this value: it **does not** identify the object itself. Thus,
48  /// whenever this object is cloned, this value is also cloned to the new
49  /// object
50  static inline int UUID = 0;

64  /// Holds information regarding the scheduling for this Node
65  struct SchedInfo {
66  /// Unique ID of the Statement/Node
67  int NodeID = -1;
68  /// TAC order
69  int TacID = 0;
70  /// Unroll factor
71  int UnrollFactor = -1;
72  /// Scope within program
73  std::vector<int> Scop = {0};
74  /// Topological order of this node
75  int FreeSched = 0;
76  /// TODO: this value should be calculated by an algorithm
77  int Plcmnt = 0;
78  /// TODO: this value stands for the scheduling order of this node; should
79  /// be also calculated when the placement
80  int Sched = 0;
81  /// If this node belongs to a reduction

85  /// If the node is NODE_OP type, then it will hold a result value
86  struct OutputInfo {
87  /// MVExpr of output
88  MVExpr *E = nullptr;
89  /// If the target variable is in memory it will be a MEM_STORE; TEMP_STORE
90  /// the other way
91  OutputType Type = MEM_STORE;
92  /// Opcode
93  MVOp MVOP;
94  /// An output node may be an operation and, therefore, may be a binary
95  /// operation
96  bool IsBinaryOp = false;
97  };

99  /// Set the output information of the given a TAC
100  OutputInfo setOutputInfo(TAC T) {
101  OutputInfo O;
102  O.E = T.getA();
103  O.Type = (T.getMVOP().isAssignment()) ? MEM_STORE : TEMP_STORE;
104  O.MVOP = T.getMVOP();
105  O.IsBinaryOp = T.getMVOP().getType() == MVOpType::CLANG_BINOP;
106  return O;
107  }

147  /// This is the unique constructor for nodes, as we will creating Nodes from
148  /// CDAG, so each TAC corresponds to a = b op c
149  Node(TAC T) {
150  this->T = NODE_OP;
151  this->MV = nullptr;
152  this->Value = T.getMVOP().toString();
153  this->O = setOutputInfo(T);
154  setSchedInfo(T);
155  this->LoopName = T.getLoopName();
156  connectInput(new Node(T.getB()));
157  if (T.getC() != nullptr) {
158  connectInput(new Node(T.getC()));
159  }
160  updateIfStore();
161  }

163  /// This is the unique constructor for nodes, as we will creating Nodes from
164  /// CDAG, so each TAC corresponds to a = b op c
165  Node(TAC T, NodeListType L) {
166  this->T = NODE_OP;
167  this->MV = nullptr;
168  this->Value = T.getMVOP().toString();
169  this->O = setOutputInfo(T);
170  setSchedInfo(T);
171  this->LoopName = T.getLoopName();
172  auto NB = findOutputNode(T.getB()->getExprStr(), L);
173  if (NB == nullptr || this->getScop()[0] != NB->getScop()[0]) {
174  connectInput(new Node(T.getB()));
175  } else {
176  connectInput(NB);
177  }
178  Node *NC = nullptr;
179  if (T.getC() != nullptr) {
180  NC = findOutputNode(T.getC()->getExprStr(), L);
181  if (NC == nullptr || this->getScop()[0] != NC->getScop()[0]) {
182  connectInput(new Node(T.getC()));
183  } else {
184  connectInput(NC);
185  }
186  }
187  updateIfStore();
188  }
189 
190  /// Update output information of the node if the type of the output is a
191  /// memory store
192  void updateIfStore() {
193  if (this->O.Type == MEM_STORE) {
194  this->getInputs().front()->T = NODE_STORE;
195  }
196  for (auto I : this->getInputs()) {
197  if (I->T == NodeType::NODE_MEM) {
198  I->SI.Scop = this->getScop();
199  I->SI.TacID = this->getTacID();
200  I->SI.UnrollFactor = this->getUnrollFactor();
201  }
202  }
203  }
