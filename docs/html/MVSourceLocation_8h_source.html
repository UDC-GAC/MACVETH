29 namespace macveth {
30 /// MVSourceLocation serves to identify the order of certain instructions,
31 /// such as reductions, initializations, etc.
32 class MVSourceLocation {
33 public:
34  /// Position specifies where the statement should be placed
35  enum Position {
36  /// Place before a certain loop nest
37  PREOUTERMOST,
38  /// Before the TAC. This is useful for initializations
39  PREORDER,
40  /// Where the TAC is originally
41  INORDER,
42  /// After the TAC. This is useful for statements that changed their location
43  /// due to transformations, e.g. loops
44  POSORDER,
45  /// Place after nested loops
46  POSOUTERMOST
47  };
48 
49  bool isPreOutermost() {
50  return this->getPosition() == MVSourceLocation::Position::PREOUTERMOST;
51  }
52 
53  bool isPreOrder() {
54  return this->getPosition() == MVSourceLocation::Position::PREORDER;
55  }
56 
57  bool isInOrder() {
58  return this->getPosition() == MVSourceLocation::Position::INORDER;
59  }
60 
61  bool isPosOrder() {
62  return this->getPosition() == MVSourceLocation::Position::POSORDER;
63  }
64 
65  bool isPosOutermost() {
66  return this->getPosition() == MVSourceLocation::Position::POSOUTERMOST;
67  }
68 
69  /// Get the order
70  int getOffset() { return this->Offset; }
71 
72  // Set order
73  void setOffset(unsigned int Offset) { this->Offset = Offset; }
74 
75  /// Get the order
76  unsigned int getOrder() { return this->Order; }
77 
78  /// Set order
79  void setOrder(unsigned int Order) { this->Order = Order; }
80 
81  /// Get the position
82  Position getPosition() { return this->P; }
83 
84  // Set position P
85  void setPosition(Position P) { this->P = P; }
86 
87  /// Empty constructor
88  MVSourceLocation() { this->P = Position::INORDER; };
89 
90  /// Constructor given position
91  MVSourceLocation(Position P) { this->P = P; };
92 
93  /// Constructor given position and order
94  MVSourceLocation(Position P, unsigned int Order) {
95  this->P = P;
96  this->Order = Order;
97  };
98 
99  /// Constructor given position, order and offset
100  MVSourceLocation(Position P, unsigned int Order, int Offset) {
101  this->P = P;
102  this->Order = Order;
103  this->Offset = Offset;
104  };
105 
106  bool operator==(MVSourceLocation M) {
107  return (M.getOffset() == this->getOffset()) &&
108  (M.getOrder() == this->getOrder()) &&
109  (M.getPosition() == this->getPosition());
110  }
111 
112 private:
113  /// This is the order where the statement or instruction should be placed
114  /// according to the original order, i.e. the TAC ID
115  unsigned int Order = 0;
116  /// This is the offset when having different locations with the same ID
117  int Offset = -1;
118  /// This is the order where the statement or instruction should be placed
119  /// regarding the order
120  Position P = Position::INORDER;
121 };
