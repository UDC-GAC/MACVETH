43 /// StmtWrapper, as its name suggests, wraps statements, its TAC representation
44 /// and the desired intrinsic translation
45 class StmtWrapper {
46 public:
47  /// Default value
48  static constexpr int UBFallback = 4;
49 
50  friend clang::Stmt;
51 
52  /// Helper structure to handle the loop information for this statement
53  struct LoopInfo {
54  /// List of dimensions declared
55  static inline std::list<std::string> DimDeclared = {};
56  /// Location of the loop itself
57  clang::SourceLocation BegLoc;
58  /// Location after the loop itself
59  clang::SourceLocation EndLoc;
60  /// Name of dimension
61  std::string Dim;
62  /// Initial value (-1 if not known)
63  long InitVal = -1;
64  /// Initial value if it can not be evaluated
65  std::string StrInitVal = "";
66  /// Upperbound value (-1 if not known)
67  long UpperBound = 0;
68  /// Comparator upperbound
69  BinaryOperator::Opcode UpperBoundComp = BinaryOperator::Opcode::BO_LT;
70  /// Upperbound as string
71  std::string StrUpperBound = "";
72  /// Stride or step
73  long Step = UBFallback;
74  /// Stride or step after unrolling
75  long StepUnrolled = UBFallback;
76  /// TODO Leftover values: if Upperbound is -1, then LeftOver as well
77  long LeftOver = -1;
78  /// Variable declared in the loop creation
79  bool Declared = false;
80  /// Unrolling factor
81  int UnrollFactor = UBFallback;
82  /// Full unroll
83  bool FullyUnrolled = false;
84  /// Location of the initialization in the loop
85  clang::CharSourceRange SRVarInit;
86  /// Location of the condition of the loop
87  clang::CharSourceRange SRVarCond;
88  /// Location of the increment in the loop
89  clang::CharSourceRange SRVarInc;
90 
91  /// Empty constructor
92  LoopInfo() {}
93 
94  /// Check if bounds are known, i.e. if values are known at compiler time
95  bool knownBounds() { return (UpperBound != -1) && (InitVal != -1); }
96 
97  /// Get epilogs given a StmtWrapper, which may contain loops within
98  static std::string getEpilogs(StmtWrapper *SWrap);
99 
100  /// Get a list of the dimensions declared in the loop
101  static std::string getDimDeclarations(std::list<std::string> DimsDeclared);
102 
103  std::string getDim() { return this->Dim; }
104 
105  /// For debugging purposes
106  std::string toString() {
107  return Dim + "; init val = " + std::to_string(InitVal) +
108  ", upperbound = " + std::to_string(UpperBound) +
109  ", step = " + std::to_string(Step) +
110  "; declared = " + std::to_string(Declared) +
111  "; stepunrolled = " + std::to_string(StepUnrolled) +
112  "; knownbounds = " + std::to_string(knownBounds());
113  }
114 
115  static void clearDims() { LoopInfo::DimDeclared.clear(); }
116  };
117

119  using LoopList = std::list<LoopInfo>;

136  /// Perform unrolling for a given statement given its unroll factor and the

139

147  bool unrollLoopList(std::list<LoopInfo> LI);
148 
149  /// Get list of statements
150  std::list<StmtWrapper *> getListStmt() { return this->ListStmt; }
151 
152  /// Get LoopInfo structure
153  LoopInfo getLoopInfo() { return this->LoopInfoStmt; }
154 
155  /// Get Clang Stmt
156  Stmt *getClangStmt() { return this->ClangStmt; };
157 
158  /// Get TAC list
159  TacListType getTacList() { return TacList; };
160 
161  /// Set TAC list in the StmtWrapper
162  void setTacList(TacListType TacList) { this->TacList = TacList; };
163 
164  /// StmtWrapper may me a "regular" statement or a for loop. This is useful
165  /// for identifying properly regions
166  bool isLoop() { return dyn_cast<ForStmt>(this->ClangStmt); }
167 
168  /// Check if StmtWrapper holds a leftover (i.e. does not hold a loop)
169  bool isLeftOver() { return !this->isLoop(); }
170 
171  /// Set the name of the loop surrounding this Stmt
172  void setInnerLoopInfo(LoopInfo InnerLoopInfo) {
173  this->InnerLoopInfo = InnerLoopInfo;
174  TacListType NewTac;
175  for (auto T : this->getTacList()) {
176  auto NT = T;
177  NT.setLoopName(InnerLoopInfo.getDim());
178  NewTac.push_back(NT);
179  }
180  this->setTacList(NewTac);
181  }
182 
183  /// Get the name of the loop surrounding this clang::Stmt
184  std::string getInnerLoopName() { return this->InnerLoopInfo.getDim(); }
185 
186  bool isInLoop() { return this->InnerLoopInfo.getDim() != ""; }
187 
188  void setNotVectorized() { this->Vectorized = false; }
189 
190  bool isVectorized() { return this->Vectorized; }
191 
192  long getScop() { return this->TacList.front().getScop(); }
193 
194 private:
195  /// Statements hold if loop
196  std::list<StmtWrapper *> ListStmt;
197  /// Statement if not loop
198  Stmt *ClangStmt;
199  /// Loop information if needed
200  LoopInfo LoopInfoStmt;
201  /// TAC list with regard to the Statement S
202  TacListType TacList;
203  /// Loop within the statement is. This only makes sense if Stmt is not a loop
204  LoopInfo InnerLoopInfo;
205  /// Has been vectorized or not
206  bool Vectorized = true;
207 };
208 } // namespace macveth
