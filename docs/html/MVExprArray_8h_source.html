130  /// Simplify expression in order to better calculate indices and so
131  void simplifyIndex() {
132  if (this->isTerminal()) {
133  return;
134  }
135  // Addition or substraction: E +/-
136  if ((this->OP == BinaryOperator::Opcode::BO_Add) ||
137  (this->OP == BinaryOperator::Opcode::BO_Sub)) {
138  if (this->LHS->isTerminal() && (this->LHS->Val == "0")) {
139  deleteLhs();
140  return;
141  }
142  if (this->RHS->isTerminal() && (this->RHS->Val == "0")) {
143  deleteRhs();
144  return;
145  }
146  }
147  // Multiplication: E * 0 or E * 1
148  if (this->OP == BinaryOperator::Opcode::BO_Mul) {
149  if ((this->LHS->isTerminal() && (this->LHS->Val == "0")) ||
150  (this->RHS->isTerminal() && (this->RHS->Val == "0"))) {
151  this->setVal("0");
152  return;
153  }
154  if (this->LHS->isTerminal() && (this->LHS->Val == "1")) {
155  deleteLhs();
156  return;
157  }
158  if (this->RHS->isTerminal() && (this->RHS->Val == "1")) {
159  deleteRhs();
160  return;
161  }
162  }
163  // Division: N / 1
164  if (this->OP == BinaryOperator::Opcode::BO_Div) {
165  if (this->RHS->isTerminal() && (this->RHS->Val == "1")) {
166  deleteRhs();
167  return;
168  }
169  }
170 
171  if (this->RHS->isTerminal() && this->LHS->isTerminal()) {
172  char *P0 = nullptr, *P1 = nullptr;
173  auto LVal = this->LHS->Val;
174  auto RVal = this->RHS->Val;
175  auto I0 = strtol(LVal.c_str(), &P0, 10);
176  auto I1 = strtol(RVal.c_str(), &P1, 10);
177  auto IsANumber = !((*P0) || (*P1));
178  if (!IsANumber) {
179  return;
180  }
181  if (this->OP == BinaryOperator::Opcode::BO_Add) {
182  auto Val = I0 + I1;
183  this->setVal(std::to_string(Val));
184  return;
185  }
186  if (this->OP == BinaryOperator::Opcode::BO_Sub) {
187  auto Val = I0 - I1;
188  this->setVal(std::to_string(Val));
189  return;
190  }
191  if (this->OP == BinaryOperator::Opcode::BO_Mul) {
192  auto Val = I0 * I1;
193  this->setVal(std::to_string(Val));
194  return;
195  }
196  if (this->OP == BinaryOperator::Opcode::BO_Div) {
197  auto Val = I0 / I1;
198  this->setVal(std::to_string(Val));
199  return;
200  }
201  }
202  }

218  this->LHS = new MVAffineIndex(Op->getLHS());
219  this->LHS->simplifyIndex();

221  this->RHS->simplifyIndex();

223  this->Val = "";

225  }

227  auto PE = P->getSubExpr();
228  // FIXME: THIS SMELLS A LOT

260  /// Update index recursively
261  bool updateIndex(int UF, std::string LL, bool SubstituteVal) {
262  if ((isTerminal()) && (LL != this->Val)) {
263  return false;
264  }
265 
266  if ((isTerminal()) && (LL == this->Val) && !Unrolled[LL] &&
267  SubstituteVal) {
268  this->Val = std::to_string(UF);
269  return false;
270  }
271 
272  if ((isTerminal()) && (LL == this->Val) && !Unrolled[LL]) {
273  this->Val = "";
274  this->LHS = new MVAffineIndex(LL);
275  this->LHS->Unrolled[LL] = true;
276  this->RHS = new MVAffineIndex(std::to_string(UF));
277  this->RHS->Unrolled[LL] = true;
278  this->OP = BinaryOperator::Opcode::BO_Add;
279  this->Unrolled[LL] = true;
280  return false;
281  }
282  if (Unrolled[LL]) {
283  return true;
284  }
285  if (this->LHS != nullptr) {
286  if (this->LHS->updateIndex(UF, LL, SubstituteVal)) {
287  this->LHS->RHS = new MVAffineIndex(std::to_string(UF));
288  } else {
289  this->simplifyIndex();
290  }
291  }
292  if (this->RHS != nullptr) {
293  if (this->RHS->updateIndex(UF, LL, SubstituteVal)) {
294  this->RHS->RHS = new MVAffineIndex(std::to_string(UF));
295  } else {
296  this->simplifyIndex();
297  }
298  }
299  return false;
300  }
