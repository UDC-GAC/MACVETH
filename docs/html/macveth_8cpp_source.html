209 // Main program: entry point for compiler
210 int main(int argc, const char **argv) {
211  if (DEBUG) {
212  llvm::sys::PrintStackTraceOnErrorSignal(argv[0]);
213  }
214 
215  if (argc <= 1) {
216  std::cout << "MACVETH: not enough arguments. Use --help option for more "
217  "information."
218  << std::endl;
219  return 0;
220  }
221 
222  // I do not like this code but I have not found any other way to do it
223  // properly... for some reason, LLVM does not like "-v" option.
224  for (int t = 0; t < argc; ++t) {
225  if ((strcmp(argv[1], "-v") == 0)) {
226  argv[t] = "--version";
227  }
228 
229  // Note for dumbs as me: const char* != std::string
230  if ((strcmp(argv[t], "--version") == 0)) {
231  std::cout << "MACVETH v" << std::to_string(MACVETH_VERSION_MAJOR) << "."
232  << std::to_string(MACVETH_VERSION_MINOR) << "."
233  << std::to_string(MACVETH_VERSION_PATCH) << std::endl;
234  break;
235  }
236  }
237 
238  // Parser for options common to all cmd-line Clang tools
239  CommonOptionsParser Op(argc, argv, MacvethCategory);
240 
241  // Utility to run a FrontendAction over a set of files
242  // * getCompilations(): contains compile cmd lines for the given source
243  // paths
244  // * getSourcePathList(): source files to run over
245  ClangTool Tool(Op.getCompilations(), Op.getSourcePathList());
246 
247  // MVOptions
248  MVOptions::OutFile =
249  OutputFile.getValue() == "" ? "macveth_output.c" : OutputFile.getValue();
250  MVOptions::InCDAGFile = CDAGInFile.getValue();
251  MVOptions::OutDebugFile = DebugFile.getValue();
252  MVOptions::ISA = ISA.getValue();
253  MVOptions::Arch = Architecture.getValue();
254  MVOptions::FMASupport = FMA.getValue();
255  MVOptions::DisableFMA = DisableFMA.getValue();
256  MVOptions::ScatterInstruction = ScatterInstruction.getValue();
257  MVOptions::Debug = Debug.getValue();
258  MVOptions::NoReformat = NoFormatCode.getValue();
259  MVOptions::Headers = !NoHeaders.getValue();
260  MVOptions::SIMDCostModel = SIMDCostModel.getValue();
261  MVOptions::IntrinsicsSVML = !NoSVML.getValue();
262  MVOptions::TargetFunc = TargetFunc.getValue();
263  MVOptions::Style = Style.getValue();
264  MVOptions::FallbackStyle = FallbackStyle.getValue();
265 
266  // Check if there are incompatible options
267  MVOptions::validateOptions();
268 
269  // Create needed files
270  Utils::initFile(MVOptions::OutFile);
271  Utils::initFile(MVOptions::OutDebugFile);
272 
273  // Runs ToolAction over all files specified in the cmd line
274  // newFrontendActionFactory returns a new FrontendActionFactory for
275  // a given type, in this case our FrontendAction, declared above
276  return Tool.run(newFrontendActionFactory<MVFrontendAction>().get());
277 }
