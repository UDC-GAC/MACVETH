39 /// Abstract class implemented by each architecture (AVX, AVX2, AVX512, etc.) to
40 /// generate specific intrinsics and calculate the associated cost for the
41 /// operations provided by the VectorAPI
42 class SIMDBackEnd {
43 public:
44  /// Types of SIMD instructions
45  enum SIMDType {
46  /// Custom function
47  VFUNC,
48  /// Load, pack memory addresses
49  VPACK,
50  /// Broadcast values
51  VBCAST,
52  /// Gather values from memory
53  VGATHER,
54  /// Set values
55  VSET,
56  /// Store values in contiguous memory
57  VSTORE,
58  /// Store values in a scattered fashion
59  VSCATTER,
60  /// Multiplication
61  VMUL,
62  /// Addition
63  VADD,
64  /// Substraction
65  VSUB,
66  /// Division
67  VDIV,
68  /// Modulo
69  VMOD,
70  /// Permutation or shuffle
71  VPERM,
72  /// Reduce operation
73  VREDUC,
74  /// Scalar or sequential operation
75  VSEQ,
76  /// If the SIMD instruction is the result of an optimization (e.g. fuse
77  /// multiply-accumulation) we will call it VOPT
78  VOPT,
79  /// Initializing some value, for instance, in reductions
80  INIT
81  };

372  /// Get the next available accumulator register
373  static std::string getNextAccmRegister(std::string V) {
374  if (AccmToReg.count(V) == 0) {
375  AccmDirty[V] = 0;
376  AccmToReg[V] = SIMDBackEnd::AccmReg++;
377  }
378  return VEC_PREFIX + std::to_string(AccmToReg.at(V));
379  }
380 
381  static bool isAccmClean(std::string V) { return !AccmDirty[V]; }
382  static void markDirtyAccm(std::string V) { AccmDirty[V] = 1; }
383 
384  /// Get the current accumulator register
385  static std::string getAccmReg(std::string V) {
386  if (AccmToReg.count(V) == 0) {
387  return "";
388  }
389  return VEC_PREFIX + std::to_string(AccmToReg.at(V));
390  }
391 
392  /// Auxiliar register unique ID

394 
395  /// Map of auxiliar registers

397 
398  /// Get the name of the auxiliar register for a operand and increment
399  static std::string getNextAuxRegister(std::string V) {
400  if (AuxReg.count(V) == 0) {
401  AuxReg[V] = SIMDBackEnd::AuxRegId++;
402  }
403  return AUX_PREFIX + std::to_string(AuxReg.at(V));
404  }
405 
406  /// Get the name of the auxiliar register for a operand
407  static std::string getAuxReg(std::string V) {
408  if (AuxReg.count(V) == 0) {
409  return "";
410  }
411  return AUX_PREFIX + std::to_string(AuxReg.at(V));
412  }
413 
414  /// Check if value has been already been mapped for an accumulator

418

424  return Name;
425  }
