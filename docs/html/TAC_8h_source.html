43 /// Class TAC: three-address-code. This abstraction is a way of representing the
44 /// Single-Statement Assignment (SSA).
45 /// This class is meant to hold structures such as:
46 /// a = b op c
47 /// Where a, b and c are Expr and op is an Opcode
48 class TAC {
49 
50 public:

59  /// Unique ID of each TAC
60  TAC::TacUUID = 0;

62  TAC::TacScop = 0;
63  /// For generating TACs
64  TAC::RegVal = 0;
65  }

105  /// Get loop name
106  std::string getLoopName() { return this->LoopName; }
107 
108  /// Get macveth operation type
109  MVOp getMVOP() { return this->MVOP; };
110 
111  /// To string method
112  std::string toString() {
113  auto C = (this->getC() != nullptr) ? ", " + this->getC()->getExprStr() : "";
114  auto Op = this->MVOP.toString();
115  Op = "t: " + this->getA()->getExprStr() + ", " +
116  this->getB()->getExprStr() + C + ", " + Op +
117  "; (loop = " + this->getLoopName() +
118  "; scop = " + std::to_string(this->getScop()) +
119  "); ID = " + std::to_string(this->getTacID()) +
120  "; UnrollID = " + std::to_string(this->getUnrollFactor());
121  return Op;
122  }
