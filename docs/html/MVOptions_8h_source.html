38 /// Costs models naming
39 enum MVSIMDCostModel {
40  /// Vectorize always, despite the not vectorized estimation cost
41  UNLIMITED,
42  /// If the sequential code vs the SIMD-zed code has a better estimation, then
43  /// MACVETH does not vectorize the code
44  CONSERVATIVE,
45  /// Region to vectorize can have parts where vectorization provides better
46  /// performance
47  AGGRESSIVE
48 };

146  /// FMA support
147  static inline bool FMASupport = false;
148  /// Disable FMA support
149  static inline bool DisableFMA = false;

152  /// Print debug information

156  /// No reformatting the final code
157  static inline bool NoReformat = false;
158  /// Generate or not macro-based code
159  static inline bool MacroCode = false;
160  /// Include headers (--no-headers)
161  static inline bool Headers = true;
162  /// SIMD Cost model
163  static inline MVSIMDCostModel SIMDCostModel = MVSIMDCostModel::CONSERVATIVE;
164  /// Intrinsics Short Vector Math Library (SVML)
165  static inline bool IntrinsicsSVML = true;

169  static MVCPUInfo::MVArch getMVArch() {
170  MVCPUInfo CInfo;
171  return CInfo.getArchitecture();
172  }
173 
174  static MVCPUInfo::MVISA getMVISA() {
175  MVCPUInfo CInfo;
176  if (CInfo.isAVX512F()) {
177  // FIXME: this is not correct
178  return MVCPUInfo::MVISA::AVX512;
179  }
180  if (CInfo.isAVX2()) {
181  return MVCPUInfo::MVISA::AVX2;
182  }
183  if (CInfo.isAVX()) {
184  return MVCPUInfo::MVISA::AVX;
185  }
186  if (CInfo.isSSE42()) {
187  return MVCPUInfo::MVISA::SSE;
188  }
189  return MVCPUInfo::MVISA::AVX2;
190  }
191 
192  /// Print options as a string for reports and so
193  static std::string toString() {
194  std::string S = "-------------------------------\n";
195  S += "\toutput file\t= " + OutFile + "\n";
196  S += "\tCDAG file\t= " + InCDAGFile + "\n";
197  S += "\toutput debug\t= " + OutDebugFile + "\n";
198  S += "\tISA and arch\t= " + MVISAStr[ISA] + ", " + MVArchStr[Arch] + "\n";
199  S += "\tFMA enabled\t= " + std::to_string(!DisableFMA) + "\n";
200  S += "\tSIMD cost model\t= " + CostModelStr[SIMDCostModel] + "\n";
201  S += "\tSVML enabled\t= " + std::to_string(IntrinsicsSVML) + "\n";
202  S += "-------------------------------\n";
203  return S;
204  }
205 
206  /// Main function to check options of the compiler
207  static void validateOptions() {
208  if (MVOptions::ISA == MVCPUInfo::MVISA::AUTODETECT) {
209  MVOptions::ISA = getMVISA();
210  }
211  if (MVOptions::Arch == MVCPUInfo::MVArch::NATIVE) {
212  MVOptions::Arch = getMVArch();
213  }
214  checkIfArchISACompatible();
215  }
216 
217  /// Dumb method to check compatibility between options
218  static void checkIfArchISACompatible() {
219  auto LArch = SupportedISAArch[MVOptions::ISA];
220  assert((std::find(LArch.begin(), LArch.end(), MVOptions::Arch) !=
221  LArch.end()) &&
222  "Incompatible architecture and ISA");
223  /// Check incompatible options:
224  assert(
225  !(MVOptions::FMASupport && MVOptions::DisableFMA) &&
226  "FMA support enabled and disabled at the same time is not possible!");
227  }
228 };
