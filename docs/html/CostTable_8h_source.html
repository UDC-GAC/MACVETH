37 /// Auxiliary class which holds the operations equivalences for the
38 /// different architectures
39 class CostTable {
40 public:
41  /// Null value in generated tables
42  static const inline std::string NullValue = "NA";
43  /// Scalar value in generated tables
44  static const inline std::string SeqValue = "seq";
45  /// Unknown cost value
46  static constexpr int UnknownCost = 10;

48  /// Latency helper class to parse properly latencies format+
49  struct Latency {
50  /// Minimum latency
51  unsigned int MinLatency = 0;
52  /// Maximum latency
53  unsigned int MaxLatency = 0;
54  /// Address latency
55  unsigned int AddrLatency = 0;
56  /// Memory latency
57  unsigned int MemLatency = 0;
58 
59  /// FIXME: Get latency
60  unsigned int getLatency() {
61  if ((MinLatency > MaxLatency) || (MinLatency == MaxLatency)) {
62  /// Is this garbage? Yyyyyes sir
63  return MinLatency;
64  }
65  return (MaxLatency + MinLatency) / 2;
66  }
67 
68  /// Get memory or address latency
69  unsigned int getLatency(std::string K) {
70  if ((K == "addr") && (AddrLatency != 0)) {
71  return AddrLatency;
72  }
73  if ((K == "mem") && (MemLatency != 0)) {
74  return MemLatency;
75  }
76  return getLatency();
77  }
78 
79  /// Empty constructor
80  Latency(){};
81 
82  /// Latency constructor based on format: # | [#;#] | [#;(#,#)] where # is a
83  /// integer
84  Latency(std::string L) {
85  if (L == NullValue) {
86  return;
87  }
88  auto open = L.find("[");
89  auto close = L.find("]") - 1;
90  if (open++ == std::string::npos) {
91  // # case
92  MinLatency = MaxLatency = std::stoi(L.c_str());
93  return;
94  } else {
95  // [#;something] case
96  std::string W;
97  auto SubStr = L.substr(open, close);
98  std::stringstream TMP(SubStr);
99  while (getline(TMP, W, ';')) {
100  auto open = W.find("(");
101  auto close = W.find(")") - 1;
102  if (open++ == std::string::npos) {
103  if (MinLatency == 0) {
104  MinLatency = std::stoi(W.c_str());
105  } else {
106  MaxLatency = std::stoi(W.c_str());
107  }
108  continue;
109  }
110  std::string Win;
111  auto SubSubStr = W.substr(open, close);
112  std::stringstream TMPin(SubSubStr);
113  while (getline(TMPin, Win, ',')) {
114  if (AddrLatency == 0) {
115  AddrLatency = std::stoi(Win.c_str());
116  } else {
117  MemLatency = std::stoi(Win.c_str());
118  }
119  }
120  }
121  }
122  };
123  };

135  /// Structure to define the table of costs for the architectures
136  struct Row {
137  /// MVOP
138  std::string MVOP = "";
139  /// Intrinsics
140  std::string Intrinsics = "";
141  /// XED iform
142  /// More documentation on:
143  std::string XED_iform = "";
144  /// ASM
145  std::string Asm = "";
146  /// CPUID
147  std::string CPUID = "";
148  /// Latency of the operation
149  Latency Lat;
150  /// Throughput of the operation: number of cycles it takes to the processor
151  /// to perform the calculations or operations of that instruction
152  double Throughput = 0;
153  /// Number of micro-instructions issued
154  int NUops = 1;
155  /// Number of ports
156  PortList NPorts = {};
157 
158  /// Empty constructor for initialization reasons
159  Row(){};
160 
161  Row(std::string MVOP, std::string Intrinsics, std::string XED_iform,
162  std::string Asm, std::string CPUID, Latency Lat, double Throughput,
163  int NUops)
164  : MVOP(MVOP), Intrinsics(Intrinsics), XED_iform(XED_iform), Asm(Asm),
165  CPUID(CPUID), Lat(Lat), Throughput(Throughput), NUops(NUops){};
166  };

194 
195  /// Get latency of operation given an architecture and the desired intrinsics
196  /// instruction, e.g. _mm256_set_pd()
197  static unsigned int getLatencyIntrin(std::string Op) {
198  if (CostTable::Tintrin[Op].Asm == SeqValue) {
199  unsigned int Lat = 0;
200  if (CostTable::Tintrin[Op].XED_iform == SeqValue) {
201  return UnknownCost;
202  }
203  std::string W;
204  std::stringstream TMP(CostTable::Tintrin[Op].XED_iform);
205  while (getline(TMP, W, ';')) {
206  Lat += CostTable::getLatency(W);
207  }
208  return Lat;
209  }
210  return CostTable::Tintrin[Op].Lat.getLatency();
211  }
212 
213  /// Get latency according to the MACVETH's operation
214  static unsigned int getLatencyMVOP(std::string Op) {
215  if (CostTable::TMVOP.find(Op) != CostTable::TMVOP.end()) {
216  return CostTable::getLatency(CostTable::TMVOP[Op].XED_iform);
217  }
218  return UnknownCost;
219  }
220 
221  static Row getIntrinRow(std::string Intrin) {
222  return CostTable::Tintrin[Intrin];
223  }
224 
225  static Row getMVOPRow(std::string MVOP) { return CostTable::TMVOP[MVOP]; }
226 
227  /// Add row to tables. This is an entry point, we will have different tables
228  /// according to the type of instruction we want to store
229  static void addRow(std::string MVOP, std::string Intrinsics, std::string ASM,
230  std::string XED_iform, std::string Lat, std::string Th,
231  std::string Uops, std::string Ports) {
232  Row R(MVOP, Intrinsics, XED_iform, ASM, "", Latency(Lat),
233  std::atof(Th.c_str()), std::atoi(Uops.c_str()));
234  if (MVOP != NullValue) {
235  CostTable::TMVOP[MVOP] = R;
236  }
237  if (Intrinsics != NullValue) {
238  CostTable::Tintrin[Intrinsics] = R;
239  }
240  if (XED_iform != SeqValue) {
241  CostTable::TXED[XED_iform] = R;
242  }
243  }
244 };
